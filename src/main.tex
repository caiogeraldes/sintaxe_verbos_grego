\chapter{Modos (em orações livres)}

\noindent Modos denotam a atitude do falante com relação ao enunciado produzido.
As orações principais ou \emph{livres} em que os modos são utilizadas podem ser
divididas entre:
\begin{inparaenum}
	\item declarativas;
	\item diretivas (comandos e ordens);
	\item desiderativas (desejos); e
	\item interrogativas (perguntas).
\end{inparaenum}
Ver~\CAGL{138s.}, sobretudo a tabela em p.~139.


\noindent Modos finitos se distinguem de modos infinitos por possuírem marca de
\textsyntax{Pessoa}.

\section{Indicativo}

\noindent Enunciados de asserção de fatos ou crenças apresentadas como fatos;
questões sobre fatos de ações passadas, presentes ou futuras. Ver~\CGG{34.5}.

\subsection{Orações principais ou \emph{livres}}
\noindent A informação mais relevante das formas
indicativas está no efeito de tempo e aspecto. Ver detalhes em~\CGG{33.11--55}.

\subsection{Orações subordinadas}
\noindent Tempo\hyp{}aspecto normalmente denota \emph{tempo relativo},
o verbo da oração principal cumprindo o papel de lastro temporal relativo ao qual o
tempo\hyp{}aspecto do verbo subordinado deve ser interpretado.

\subsection{Usos especiais}

\subsubsection{Interrogativas expressando comandos urgentes ou pedidos}
\noindent οὐ + futuro indicativo de 2ª pessoa; negado por οὐ μή, que denota proibições
urgentes. Ver~\CGG{38.32},~\SSVCG{9}.


\subsubsection{Interrogativas expressando pedidos ou sugestões}
\noindent (τί) οὐ + presente\slash{}aoristo indicativo de 1ª ou 2ª pessoa.
Ver~\CGG{38.33},~\SSVCG{5.2, 8.3.3}.

\subsubsection{Indicativo modal}
\noindent Tempos secundários no indicativo podem indicar
eventos contrafactuais ou modalidade \emph{irrealis}. Normalmente acompanhados
por ἄν, mas exceções existem sobretudo com verbos impessoais modais,
ver~\SSVCG{4, 8.1}. Ver também a discussão sobre condicionais.
Outra construção modal com indicativos secundários denota desejos não
realizados, sem ἄν, sempre seguindo εἰ γάρ ou εἴθε, ver~\SSVCG{8.2}.
Detalhes em~\CGG{34.15--18}.

\section{Subjuntivo}

\noindent Nas construções a seguir, tempo\hyp{}aspecto denota apenas
\emph{aspecto}, sendo o perfeito subjuntivo raro nelas.
Salvo em grego homérico, a partícula ἄν não é empregada com subjuntivos de
orações \emph{livres}, mas ver~\GH{II}{311--3} sobre o emprego do subjuntivo +
ἄν/κε{(ν)} em orações \emph{livres} em Homero.

\subsection{Hortativo ou jussivo}\label{subjhort}
\noindent Em primeira pessoa, geralmente plural, o subjuntivo denota comandos positivos e
negativos (com μή). A segunda ou terceira pessoa do subjuntivo aoristo produz o
sentido de proibição. Ver~\CGG{34.6-7},~\SSVCG{13.1}.

\subsection{Deliberativo ou dubitativo}
\noindent Em sentenças interrogativas, a primeira pessoa do subjuntivo denota a incerteza
do falante sobre um certo estado de coisas. Ver~\CGG{34.8},~\SSVCG{13.2}.

\subsection{Negação enfática}
\noindent οὐ μή + subjuntivo expressa forte certeza de que
algum estado de coisa não ocorrerá \emph{no futuro}. Ver~\CGG{34.9},~\SSVCG{20
	N.~3--4}.

\subsection{Construções de medo}
\noindent μή + subjuntivo pode ocorrer em uma oração \emph{livre} como se
subordinada a um verbo de medo, expressando uma asserção ansiosa ou cautelosa,
negada por μὴ οὐ. Praticamente restrita a Platão. Ver~\CGG{34.10},~\SSVCG{20
	N.~3}.

\section{Optativo}

\subsection{Optativo potencial (+ἄν)}

\noindent Formas do optativo + ἄν em orações \emph{livres}
denotam estados de coisas que o falante considera que podem hipoteticamente
acontecer ou que ele busca afirmar com cuidado (por poderem ser falsas).
A negação é construída com οὐ e expressa uma negação enfática.
Construções de optativo potencial em segunda pessoa denotam comandos ou pedidos
enunciados de maneira cautelosa.
Formas de primeira pessoa do optativo potencial podem denotar que o falante
se dá o direito de executar uma certa ação cautelosamente ou aceita um comando
ou pedido.
Tempo\hyp{}aspecto é puramente aspectual.
Detalhes em~\CGG{34.13} e~\SSVCG{14.2}.\footnote{Notar que em sentenças
	interrogativas, o optativo potencial denota que o falante interroga sobre a
	\emph{possibilidade} de um estado de coisas ou detalhes sobre o estado de coisas
	interrogado. Ver~\SSVCG{14.2.3}.}

\subsection{Optativo cupitivo (-ἄν)}
\noindent Formas do optativo - ἄν em orações \emph{livres}
denotam desejos e são frequentemente precedidas por εἴθε, εἰ γάρ ou ὡς. Negação
em μή.
Tempo\hyp{}aspecto é puramente aspectual.
Detalhes em~\CGG{34.14},~\SSVCG{14.1}.

\section{Imperativo}

\noindent Usado em ordens e proibições, negado por μή.
Proibições de segunda pessoa que necessitem de expressão de aoristo normalmente
são construídas com subjuntivos, ver~\ref{subjhort} acima.
Por vezes expressa permissão.
Detalhes em~\CGG{34.19-21},~\SSVCG{15}.
Tempo\hyp{}aspecto expressa apenas aspectualidade,~\CGG{38.20},~\SSVCG{16.2}.
